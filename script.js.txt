(function () {
  const prefersReducedMotion = window.matchMedia("(prefers-reduced-motion: reduce)").matches;

  // Reveal on scroll
  const revealItems = document.querySelectorAll(".reveal");
  if (!prefersReducedMotion && "IntersectionObserver" in window) {
    const observer = new IntersectionObserver(
      (entries) => {
        entries.forEach((entry, index) => {
          if (entry.isIntersecting) {
            entry.target.style.setProperty("--delay", `${Math.min(index * 60, 220)}ms`);
            entry.target.classList.add("is-visible");
            observer.unobserve(entry.target);
          }
        });
      },
      { threshold: 0.14 }
    );

    revealItems.forEach((item) => observer.observe(item));
  } else {
    revealItems.forEach((item) => item.classList.add("is-visible"));
  }

  // Header hide/reveal on scroll down/up
  const header = document.querySelector(".site-header");
  let lastScrollY = window.scrollY;

  window.addEventListener("scroll", () => {
    const currentY = window.scrollY;
    if (currentY > lastScrollY && currentY > 120) {
      header.classList.add("is-hidden");
    } else {
      header.classList.remove("is-hidden");
    }
    lastScrollY = currentY;
  });

  // Magnetic buttons on pointer devices
  if (!prefersReducedMotion && window.matchMedia("(hover: hover) and (pointer: fine)").matches) {
    document.body.classList.add("cursor-active");

    const magneticItems = document.querySelectorAll(".magnetic");

    magneticItems.forEach((item) => {
      item.addEventListener("mousemove", (event) => {
        const rect = item.getBoundingClientRect();
        const x = event.clientX - rect.left - rect.width / 2;
        const y = event.clientY - rect.top - rect.height / 2;

        item.style.transform = `translate(${x * 0.12}px, ${y * 0.12}px)`;
      });

      item.addEventListener("mouseleave", () => {
        item.style.transform = "";
      });
    });

    // Custom cursor
    const cursor = document.querySelector(".cursor-ring");
    let cursorX = window.innerWidth / 2;
    let cursorY = window.innerHeight / 2;
    let ringX = cursorX;
    let ringY = cursorY;

    window.addEventListener("mousemove", (e) => {
      cursorX = e.clientX;
      cursorY = e.clientY;
      cursor.classList.add("is-visible");
    });

    const hoverTargets = document.querySelectorAll("a, button, .project-card, .about-stat");

    hoverTargets.forEach((target) => {
      target.addEventListener("mouseenter", () => cursor.classList.add("is-hovering"));
      target.addEventListener("mouseleave", () => cursor.classList.remove("is-hovering"));
    });

    function animateCursor() {
      ringX += (cursorX - ringX) * 0.18;
      ringY += (cursorY - ringY) * 0.18;
      cursor.style.transform = `translate(${ringX}px, ${ringY}px) translate(-50%, -50%)`;
      requestAnimationFrame(animateCursor);
    }

    animateCursor();

    // Project glow tracking
    document.querySelectorAll(".project-card").forEach((card) => {
      card.addEventListener("mousemove", (e) => {
        const rect = card.getBoundingClientRect();
        const mx = ((e.clientX - rect.left) / rect.width) * 100;
        const my = ((e.clientY - rect.top) / rect.height) * 100;
        card.style.setProperty("--mx", `${mx}%`);
        card.style.setProperty("--my", `${my}%`);
      });
    });

    // Parallax
    const parallaxItems = document.querySelectorAll("[data-parallax]");
    window.addEventListener("mousemove", (e) => {
      const x = (e.clientX / window.innerWidth - 0.5) * 2;
      const y = (e.clientY / window.innerHeight - 0.5) * 2;

      parallaxItems.forEach((item) => {
        const strength = Number(item.dataset.parallax || 10);
        item.style.transform = `translate(${x * strength}px, ${y * strength}px)`;
      });
    });

    window.addEventListener("mouseleave", () => {
      document.querySelectorAll(".magnetic").forEach((item) => {
        item.style.transform = "";
      });
      document.querySelectorAll("[data-parallax]").forEach((item) => {
        item.style.transform = "";
      });
    });
  }

  // Graceful fallback for missing portrait image
  const portraitImg = document.querySelector(".portrait-card img");
  const fallback = document.querySelector(".portrait-fallback");

  if (portraitImg && fallback) {
    portraitImg.addEventListener("error", () => {
      portraitImg.style.display = "none";
      fallback.style.display = "flex";
    });
  }
})();
